CREATE TABLE public."ClientLetter"
(
    id uuid NOT NULL DEFAULT gen_random_uuid(),
    name character(200)[] COLLATE pg_catalog."default" NOT NULL,
    email character(200)[] COLLATE pg_catalog."default" NOT NULL,
    phone character(100)[] COLLATE pg_catalog."default" NOT NULL,
    message character(500)[] COLLATE pg_catalog."default" NOT NULL,
    date timestamp without time zone NOT NULL DEFAULT now(),
    CONSTRAINT "ClientLetter_pkey" PRIMARY KEY (id)
)
